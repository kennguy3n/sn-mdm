# Runtime dependencies for the sn-mdm crawl layer.
# Pinned to ranges that have shipped together for >12 months so the
# CI snapshot stays reproducible without lock-file drift.

# HTTP + HTML parsing
requests>=2.32,<3
beautifulsoup4>=4.12,<5
lxml>=5.2,<6

# PDF transcript extraction (BCG, McKinsey, Deutsche Bank,
# Microsoft Cyber transcripts ship as PDFs).
pdfminer.six>=20240706

# Headless browser transport for publishers whose discovery /
# episode pages are React / Next.js / Drupal SPAs (RBC
# Disruptors) or sit behind a WAF rule that rejects plain
# requests-shaped TLS fingerprints (WEF Radio Davos). Used only
# by the ``crawl.crawlers._browser`` module; crawlers that don't
# need a headless browser never import Playwright at all.
#
# A successful Playwright install also requires the matching
# Chromium build, which is downloaded out-of-band by
# ``python -m playwright install chromium`` (added to the repo
# setup steps). CI caches the resulting browser binaries.
playwright>=1.41,<2

# Content-hash for the chunk dedup gate. Pinned as a hard
# dependency (not an optional fallback to SHA-256) so a host
# without the blake3 wheel cannot produce a different digest from
# a host with it — which would silently break the
# ``_load_known_hashes`` short-circuit in ``crawl/pipeline.py``
# when the dependency state changes between runs. The Rust ingest
# path uses BLAKE3 unconditionally, so this also keeps the Python
# and Rust digests byte-for-byte identical.
blake3>=0.4,<1

# robots.txt parsing
urllib3>=2.2,<3

# Toml parsing — Python 3.11+ ships tomllib in the stdlib, but the
# package still needs to support 3.10 for older host environments.
tomli>=2.0,<3; python_version < "3.11"

# Test framework
pytest>=8.2,<9

# Lint / format
ruff>=0.6,<1
